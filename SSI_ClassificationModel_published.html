# create training and test data 
library(caTools)
set.seed(123)
split = sample.split(ssi_2019$HasSsi, SplitRatio = 0.75)
training_set = subset(ssi_2019, split == TRUE)
test_set = subset(ssi_2019, split == FALSE)
# Fitting  Naive Bayes  classifier to the Training set
library(e1071)
classifier = naiveBayes(x = training_set,
                        y = training_set$HasSsi)
# Predicting the Test set results
y_pred = predict(classifier, newdata = test_set)
# Making the Confusion Matrix
library(caret)
confusionMatrix(factor(y_pred),
                factor(test_set$HasSsi))

rocplot=function(pred,truth, ...){
  predob=prediction(pred,truth)
  perf=performance(predob,"tpr","fpr")
  auc=performance(predob,"auc")@y.values
  plot(perf,main = auc)
  
}
